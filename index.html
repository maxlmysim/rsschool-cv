function isPrime(num) {
  if (num === 2 || num === 3) return true
  if (num < 2 || num % 2 === 0 || num % 3 === 0) return false;
  for (let i = 1; i <= Math.sqrt(num); i++) {
    if (num % (6 * i + 1) === 0 && (num !== (6 * i + 1))) return false;
    if (num % (6 * i - 1) === 0 && (num !== (6 * i - 1))) return false;
    }
  return true;
}
